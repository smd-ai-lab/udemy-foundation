🔌 Middleware

Vue d'ensemble

Le middleware est le mécanisme central de personnalisation dans LangChain v1.0. Il permet d'intercepter, modifier ou étendre le comportement de l'agent à différentes étapes.

python

from langchain.agents.middleware import Middleware

from typing import Any, Dict

class MonMiddleware(Middleware):

    async def before_agent(self, state: Dict[str, Any]) -> Dict[str, Any]:

        print("Avant l'exécution de l'agent")

        return state

    

    async def after_agent(self, state: Dict[str, Any]) -> Dict[str, Any]:

        print("Après l'exécution de l'agent")

        return state

Middleware Préconstruit

LangChain fournit plusieurs middlewares prêts à l'emploi :

python

from langchain.agents.middleware import (

    SummarizationMiddleware,

    HumanInTheLoopMiddleware,

    ToolRetryMiddleware,

    RateLimitMiddleware

)

agent = create_agent(

    model=model,

    tools=outils,

    middleware=[

        # 1. Résumé automatique des longues conversations

            model="gpt-4o-mini",

            trigger={"tokens": 1000}  # Déclenche à 1000 tokens

        ),

        

        # 2. Validation humaine avant actions sensibles

        HumanInTheLoopMiddleware(

            interrupt_on={"envoyer_email": {"approval_required": True}}

        ),

        

        # 3. Réessai automatique en cas d'échec

        ToolRetryMiddleware(

            max_retries=3,

            backoff_factor=2.0

        ),

        

        # 4. Limitation du taux de requêtes

        RateLimitMiddleware(

            requests_per_minute=60,

            requests_per_day=1000

        )

    ]

)

Middleware Personnalisé

Exemple d'un middleware de logging :

python

import time

from langchain.agents.middleware import Middleware

from typing import Any, Dict, Optional

class LoggingMiddleware(Middleware):

    def __init__(self, log_file: str = "agent.log"):

        self.log_file = log_file

    

    async def before_agent(self, state: Dict[str, Any]) -> Dict[str, Any]:

        self.start_time = time.time()

        user_message = state["messages"][-1]["content"]

        self._log(f"Question utilisateur: {user_message}")

        return state

    

    async def after_agent(self, state: Dict[str, Any]) -> Dict[str, Any]:

        duration = time.time() - self.start_time

        response = state["messages"][-1]["content"]

        self._log(f"Réponse (en {duration:.2f}s): {response[:100]}...")

        return state

    

    def _log(self, message: str):

        with open(self.log_file, "a") as f:

            f.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - {message}\n")

# Utilisation

agent = create_agent(

    model=model,

    tools=outils,

    middleware=[LoggingMiddleware()]

)

⚙️ Advanced Usage

1. Guardrails

Sécuriser les réponses de l'agent :

python

from langchain.agents.middleware import GuardrailMiddleware

def verification_securite(messages: list) -> bool:

    """Vérifie si le message est approprié"""

    interdit = ["mots", "interdits", "sensibles"]

    dernier_message = messages[-1]["content"].lower()

    return not any(mot in dernier_message for mot in interdit)

agent = create_agent(

    model=model,

    middleware=[

        GuardrailMiddleware(

            check=verification_securite,

            on_violation="block",  # ou "warn", "override"

            fallback_response="Je ne peux pas répondre à cette question."

        )

    ]

)

6. Retrieval (RAG - Retrieval-Augmented Generation)

Augmentation de la génération par recherche documentaire :

python

from langchain.vectorstores import Chroma

from langchain.embeddings import OpenAIEmbeddings

from langchain.document_loaders import TextLoader

from langchain.text_splitter import RecursiveCharacterTextSplitter

# 1. Chargement et indexation des documents

docs = TextLoader("ma_base_connaissances.txt").load()

splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)

chunks = splitter.split_documents(docs)

vectorstore = Chroma.from_documents(

    chunks, 

    OpenAIEmbeddings(),

    collection_name="ma_base"

)

# 2. Intégration dans l'agent

from langchain.agents.middleware import RetrievalMiddleware

agent = create_agent(

    model=model,

    middleware=[

        RetrievalMiddleware(

            vectorstore=vectorstore,

            k=5,  # Récupère les 5 documents les plus pertinents

            strategy="relevance_score",

            threshold=0.7

        )

    ]

)

7. Long-term Memory (Mémoire Long Terme)

Maintien d'informations persistantes entre les sessions :

python

from langgraph.store import RedisStore

from langchain.agents.middleware import LongTermMemoryMiddleware

# Stockage persistant (Redis par exemple)

store = RedisStore(host="localhost", port=6379)

agent = create_agent(

    model=model,

    middleware=[

        LongTermMemoryMiddleware(

            store=store,

            memory_types=["user_preferences", "facts", "history"],

            auto_save=True,

            retrieval_strategy="semantic",  # ou "recency", "importance"

            max_memories=50

        )

    ]

)

# L'agent se souvient des préférences de l'utilisateur

}, config={"configurable": {"user_id": "alice@email.com"}})
